Interface ColorLoggerExt

Provides an extended set of logging methods to explicitly control options.

interface ColorLoggerExt {
    debugCompact(...msg): string;
    debugNoColor(...msg): string;
    debugRaw(...msg): string;
    debugTime(...msg): string;

    traceCompact(...msg): string;
    traceNoColor(...msg): string;
    traceRaw(...msg): string;
    traceTime(...msg): string;
    verboseCompact(...msg): string;
    verboseNoColor(...msg): string;
    verboseRaw(...msg): string;
    verboseTime(...msg): string;
    warnCompact(...msg): string;
    warnNoColor(...msg): string;
    warnRaw(...msg): string;
    warnTime(...msg): string;
}
