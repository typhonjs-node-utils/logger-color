@typhonjs-utils/logger-color

    Interface ColorLoggerIs

    Provides the is API allowing compact checks for log level conditional statements.

    Example

    if (logger.is.verbose) {
       // The logger is configured for verbose logging.
    }
    interface ColorLoggerIs {
        get debug(): boolean;
        get error(): boolean;
        get fatal(): boolean;
        get info(): boolean;
        get trace(): boolean;
        get verbose(): boolean;
        get warn(): boolean;
    }
